\section{Computation of \texorpdfstring{$\Psi_4$}{Psi4}}
The waveforms we plot are often given in terms of what is knoiwn as the Newman penrose scalar this is one way to decompose the graviotational waves that come off of our systesm. 
In this appendix, we introduce the Newman--Penrose scalar $\Psi_4$ and describe how it is computed in numerical relativity.

We begin by defining a tetrad. We take the timelike vector to be normal to the spacelike hypersurface. In the asymptotic region, we may construct a set of spatial vectors resembling spherical coordinates and then orthonormalize them using the Gram--Schmidt procedure.

\begin{equation}
\begin{aligned}
    u^a &= (0, x, y, z),\\
    v^a &= (0, xz, yz, -x^2 - y^2),\\
    w^a &= (0, -y, x, 0).
\end{aligned}
\end{equation}

Using the spatial metric $h_{ij}$, we orthonormalize them with respect to it: In particular we define new orthonormal space like vectors as:
\begin{equation}
\begin{aligned}
    {}^{(1)}e^a &= \frac{u^i}{\|u\|},\\
    {}^{(2)}e^a &= \frac{v^i - \langle {}^{(1)}e, v \rangle \, {}^{(1)}e^i}{\|v - \langle {}^{(1)}e, v \rangle {}^{(1)}e\|},\\
    {}^{(3)}e^a &= \frac{w^i - \langle {}^{(1)}e, w \rangle {}^{(1)}e^i - \langle {}^{(2)}e, w \rangle {}^{(2)}e^i}{\|w - \langle {}^{(1)}e, w \rangle {}^{(1)}e - \langle {}^{(2)}e, w \rangle {}^{(2)}e\|}.
\end{aligned}
\end{equation}
where we define the inner product and the norm as
\begin{equation}
\langle u, v \rangle \equiv h_{ij} u^i v^j
\end{equation}
\begin{equation}
\|u\| \equiv \sqrt{\langle u, u \rangle}
\end{equation}
We identify ${}^{(1)}e^a = r^a$, ${}^{(2)}e^a = \theta^a$, and ${}^{(3)}e^a = \phi^a$. Using these, we construct the null tetrad:

\begin{equation}
\begin{aligned}
    \ell^a &= \frac{1}{\sqrt{2}}(n^a + r^a),\\
    \tilde{n}^a &= \frac{1}{\sqrt{2}}(n^a - r^a),\\
    m^a &= \frac{1}{\sqrt{2}}(\theta^a + i \phi^a),\\
    \bar{m}^a &= \frac{1}{\sqrt{2}}(\theta^a - i \phi^a).
\end{aligned}
\end{equation}
    \section{Null Tetrads and Their Properties}

A null tetrad is a set of four vectors 
\[
\{ \ell^a, n^a, m^a, \bar{m}^a \}
\]
forming a basis for the tangent space at each point in spacetime. Here $\ell^a$ and $n^a$ are real null vectors, while $m^a$ is a complex null vector and $\bar{m}^a$ is its complex conjugate. This formalism is central to the Newman--Penrose approach to general relativity \cite{Wald1984,Carroll2004}.

\subsection{Null Conditions}

Each vector in the tetrad is null:
\begin{equation}
\ell^a \ell_a = 0, \quad n^a n_a = 0, \quad m^a m_a = 0, \quad \bar{m}^a \bar{m}_a = 0.
\end{equation}

\subsection{Orthogonality Relations}

The tetrad vectors satisfy the following orthogonality conditions:
\begin{equation}
\ell^a m_a = \ell^a \bar{m}_a = n^a m_a = n^a \bar{m}_a = 0.
\end{equation}

\subsection{Normalization}

The only non-vanishing inner products are
\begin{equation}
\ell^a n_a = -1, \qquad m^a \bar{m}_a = 1,
\end{equation}
where we have assumed the metric signature $(-,+,+,+)$.

\section{The Weyl Tensor}

We shall mostly be concerned with tensors in four dimensions or less. 
The algebraic identities lead to the following special cases for the curvature tensor:

\begin{enumerate}
    \item If $n = 1$, $R_{abcd} = 0$.
    \item If $n = 2$, $R_{abcd}$ has one independent component --- essentially $R$.
    \item If $n = 3$, $R_{abcd}$ has six independent components --- essentially $R_{ab}$.
    \item If $n = 4$, $R_{abcd}$ has twenty independent components --- ten of which are given by $R_{ab}$ and the remaining ten by the Weyl tensor.
\end{enumerate}

The \textbf{Weyl tensor}, or \textbf{conformal tensor}, $C_{abcd}$ is defined in $n$ dimensions ($n \geq 4$) by
\begin{equation}
\begin{aligned}
C_{abcd} &= R_{abcd} 
+ \frac{1}{n-2} \left( g_{ad} R_{cb} + g_{bc} R_{da} - g_{ac} R_{db} - g_{bd} R_{ca} \right) \\
&\quad + \frac{1}{(n-1)(n-2)} \left( g_{ac} g_{db} - g_{ad} g_{cb} \right) R.
\end{aligned}
\end{equation}

Thus, in four dimensions, this becomes
\begin{equation}
\begin{aligned}
C_{abcd} &= R_{abcd} 
+ \frac{1}{2} \left( g_{ad} R_{cb} + g_{bc} R_{da} - g_{ac} R_{db} - g_{bd} R_{ca} \right) \\
&\quad + \frac{1}{6} \left( g_{ac} g_{db} - g_{ad} g_{cb} \right) R.
\end{aligned}
\end{equation}

The Weyl tensor possesses the same symmetries as the Riemann tensor, namely,
\begin{equation}
C_{abcd} = -C_{abdc} = -C_{bacd} = C_{cdab},
\end{equation}
and
\begin{equation}
C_{abcd} + C_{adbc} + C_{acdb} \equiv 0.
\end{equation}

 The Newman--Penrose scalar $\Psi_4$ is defined as

\begin{equation}
    \Psi_4 = C_{abcd}\, n^a \bar{m}^b n^c \bar{m}^d.
\end{equation}
All parts vanish except for the Reimann tensor term. we will now decompose the this and look at several parts in turn. 
starting with:
\begin{equation}
    \Psi_4 = R_{abcd}\, n^a \bar{m}^b n^c \bar{m}^d.
\end{equation}

Expanding the projection onto the hypersurface, we obtain

\begin{equation}
\begin{aligned}
    \Psi_4 &= \perp(R_{abcd} n^a n^c)\, \bar{m}^b \bar{m}^d
    - \perp(R_{abcd} n^a)\, r^c \bar{m}^b \bar{m}^d.
\end{aligned}
\end{equation}

Using the Gauss, Codazzi, and Ricci relations, we have

\begin{equation}
    \perp R_{abcd} = {}^{(3)}R_{abcd} + K_{ac}K_{bd} - K_{ad}K_{bc},
\end{equation}

\begin{equation}
    \perp(R_{abcd} n^a) = D_c K_{bd} - D_d K_{bc},
\end{equation}

\begin{equation}
    \perp(R_{abcd} n^a n^c) = \mathcal{L}_n K_{bd} + K_{bc}K^c{}_d + \frac{1}{\alpha} D_b D_d \alpha.
\end{equation}

Substituting these relations, we find

\begin{equation}
\begin{aligned}
    \Psi_4 &= \left({}^{(3)}R_{bd} + K K_{bd} - K_{bc}K^c{}_d \right)\bar{m}^b \bar{m}^d \\
    &\quad + (D_c K_{bd} - D_d K_{bc})\, \bar{m}^b \bar{m}^d r^c.
\end{aligned}
\end{equation}

Expressing these quantities in terms of the BSSN variables, we use

\begin{equation}
    K_{ab} = e^{4\phi} \left( \tilde{A}_{ab} + \frac{1}{3}\tilde{\gamma}_{ab} K \right).
\end{equation}

The full expressions in terms of BSSN variables are lengthy; here we present the final result. The real part is

\begin{equation}
\begin{aligned}
\mathcal{R}(\Psi_4) &= {}^{(3)}R_{ab} M^{ab} - 4\pi G S_{ab} M^{ab} \\
&\quad + \left(\chi^{-1}\tilde{A}_{ab} r^a r^b + \frac{1}{3}K\right)\chi^{-1}\tilde{A}_{cd} M^{cd} \\
&\quad - \chi^{-2}\left[(\tilde{A}_{ab} r^a m^b)^2 - (\tilde{A}_{ab} r^a \bar{m}^b)^2\right] + \cdots
\end{aligned}
\end{equation}

and the imaginary part is

\begin{equation}
\begin{aligned}
\mathcal{I}(\Psi_4) &= - \Big[ {}^{(3)}R_{ab} N^{ab} - 4\pi G S_{ab} N^{ab} + \cdots \Big].
\end{aligned}
\end{equation}
